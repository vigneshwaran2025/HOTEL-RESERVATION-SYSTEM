# Hotel Reservation System

rooms = {
    101: None,
    102: None,
    103: None,
    104: None,
    105: None
}

# Show available rooms
def show_rooms():
    print("\nRoom Status:")
    for room, guest in rooms.items():
        if guest is None:
            print(f"Room {room}: Available")
        else:
            print(f"Room {room}: Booked by {guest}")

# Book a room
def book_room():
    room_no = int(input("Enter room number to book: "))
    
    if room_no in rooms:
        if rooms[room_no] is None:
            name = input("Enter customer name: ")
            rooms[room_no] = name
            print("Room booked successfully!")
        else:
            print("Room already booked!")
    else:
        print("Invalid room number!")

# Cancel booking
def cancel_booking():
    room_no = int(input("Enter room number to cancel: "))
    
    if room_no in rooms:
        if rooms[room_no] is not None:
            rooms[room_no] = None
            print("Booking canceled successfully!")
        else:
            print("Room is already empty!")
    else:
        print("Invalid room number!")

# Main menu
while True:
    print("\n--- Hotel Reservation System ---")
    print("1. Show Rooms")
    print("2. Book Room")
    print("3. Cancel Booking")
    print("4. Exit")
    
    choice = input("Enter choice: ")

    if choice == '1':
        show_rooms()
    elif choice == '2':
        book_room()
    elif choice == '3':
        cancel_booking()
    elif choice == '4':
        print("Thank you!")
        break
    else:
        print("Invalid choice! Try again.")
